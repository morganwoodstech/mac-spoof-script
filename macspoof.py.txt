#!/usr/bin/env python

# The subprocess module allows you to spawn new processes, connect to their input/output/error pipes, and obtain
# their return codes.
import subprocess

# Retrieves user input for network interface and mac_address in example format of eth0 / aa:bb:cc:dd:ee:ff
# Contains python3 functions. For python2 compatibility change function to raw_input()
interface = input("Enter network interface >  ")
new_mac = input("Enter new MAC address > ")

# Prints conformation regarding user input
print("[+] Changing MAC address for " + interface + " to " + new_mac)

# module.function(arg1, arg2) runs shell commands with previously defined parameters
subprocess.call("ifconfig " + interface + " down", shell=True)
subprocess.call("ifconfig " + interface + " hw ether " + new_mac, shell=True)
subprocess.call("ifconfig " + interface + " up", shell=True)
